<?xml version="1.0" encoding="utf-8"?>
<s:VGroup xmlns:fx="http://ns.adobe.com/mxml/2009" 
         xmlns:s="library://ns.adobe.com/flex/spark" 
         xmlns:mx="library://ns.adobe.com/flex/mx" 
         xmlns:presentation="com.deCampredon.sampleContact.presentation.*" 
         xmlns:spicefactory="http://www.spicefactory.org/parsley"
         xmlns:ns="http://ns.adobe.com/cairngorm"
         horizontalAlign="center" 
         verticalAlign="middle">
    <fx:Declarations>
        <presentation:ContactFormPM id="pm"/>
        <spicefactory:FastInject type="{ContactFormPM}" property="pm"/>
        <ns:ObserveValue source="{pm.showUnSavedChangeWarning}" 
                         handler="showSelectedContactChangeWarning"
                         sendArgumentsToHandler="false" 
                         value="true"/>
    </fx:Declarations>
    <fx:Script>
        <![CDATA[
            import mx.controls.Alert;
            import mx.events.CloseEvent;
            private function showSelectedContactChangeWarning():void {
                Alert.show(
                    "Do you want to abandon your changes ?",
                    "Confirm",
                    Alert.YES | Alert.NO, 
                    this, 
                    handleAlertClose);
            }
            
            private function handleAlertClose(event:CloseEvent):void {
                if(event.detail == Alert.YES)
                    pm.confirmSelectedContactChange();
                else
                    pm.cancelSelectedContactChange();
            }
        ]]>
    </fx:Script>
    <mx:Form>
        <mx:FormItem label="First Name : ">
            <s:TextInput text="@{pm.selectedContact.firstName}" />
        </mx:FormItem>
        <mx:FormItem label="Last Name : ">
            <s:TextInput text="@{pm.selectedContact.lastName}" />
        </mx:FormItem>
        <mx:FormItem label="Email : ">
            <s:TextInput text="@{pm.selectedContact.email}" />
        </mx:FormItem>
        <mx:FormItem label="Phone Number : ">
            <s:TextInput text="@{pm.selectedContact.phoneNumber}" />
        </mx:FormItem>
    </mx:Form>
    <s:HGroup>
        <s:Button label="Save" enabled="{pm.hasUnsavedChange}" click="pm.saveChange()" />
        <s:Button label="Cancel" enabled="{pm.hasUnsavedChange}" click="pm.revertChange()" />
    </s:HGroup>
</s:VGroup>
