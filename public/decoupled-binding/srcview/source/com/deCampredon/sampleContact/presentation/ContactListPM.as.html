package com.deCampredon.sampleContact.presentation
{
    import com.deCampredon.sampleContact.controls.events.ContactEvent;
    import com.deCampredon.sampleContact.domain.vo.Contact;
    
    import flash.events.EventDispatcher;
    
    import mx.collections.IList;
    
    [Event("getContacts",type="com.deCampredon.sampleContact.controls.events.ContactEvent")]
    
    [ManagedEvents("getContacts")]

    public class ContactListPM extends EventDispatcher
    {
        
        
        [Bindable][Subscribe("contactList")]
        public var contactList:IList;
        
        public var _selectedContact:Contact;
        
        [Bindable][PublishSubscribe("selectedContact")]
        public function get selectedContact():Contact {
            return _selectedContact;
        }
        public function set selectedContact(value:Contact):void {
            _selectedContact = value;
            trace("gne");
        }
        
        public function ContactListPM() {
            super();
        }
        
        public function getContactList():void {
            dispatchEvent(new ContactEvent(ContactEvent.GET_CONTACTS));
        }
    }
}
