package com.deCampredon.sampleContact.presentation
{
    import com.deCampredon.sampleContact.domain.vo.Contact;
    
    import mx.collections.IList;

    [Bindable]public class MessageBoardPM
    {
        
        [Subscribe("contactList")]
        public var contactList:IList;
        
        public var contact:Contact;
        
        public var object:String;
        
        public var message:String;
        
        public var selectedContact:Contact;
        
        public var showConfirmMessageSend:Boolean;
        
        public function sendMessage():void {
            contact = null;
            object = null;
            message = null;
            showConfirmMessageSend = true;
        }
    }
}
