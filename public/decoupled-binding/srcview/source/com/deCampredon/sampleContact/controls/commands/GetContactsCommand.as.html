package com.deCampredon.sampleContact.controls.commands
{
    import com.deCampredon.sampleContact.controls.events.ContactEvent;
    import com.deCampredon.sampleContact.domain.ContactDomain;
    import com.deCampredon.sampleContact.services.ContactService;
    
    import mx.collections.ArrayList;
    import mx.rpc.AsyncToken;
    import mx.rpc.Fault;

    public class GetContactsCommand
    {
        [Inject]public var service:ContactService;
        
        [Inject]public var domain:ContactDomain;
        
        public function execute (event:ContactEvent) : AsyncToken {
            return service.getContacts();
        }
        
        public function result (result:Array) : void {
            domain.contactList = new ArrayList(result);
        }
        
        public function error (fault:Fault) : void {

        }
    }
}
